argc记录了用户在运行程序的命令行中输入的参数的个数。

    {% highlight c %}
    int main(int argc, char* argv[]) 
    { 
        for(int i = 0; i < argc; i++) 
        {
            printf("%s \n", argv[i]);
        }
    
        return 0; 
    }      
    {% endhighlight %}     

另外一种main参数方式比较少用，那就是`main(int argc, char** argv, char** envp)`，
多了一个envp参数。这个参数是用来获取系统参数。 

    {% highlight c %}
    int main(int argc, char* argv[], char* envp[])
    {
        char** ppszEnv = envp;
        while (*ppszEnv)
        {
            printf("%s\n", *ppszEnv);
            ppszEnv++;
        }
     
        return 0;
    }
    {% endhighlight %}
